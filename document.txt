// a document is composed of elements
document   : EOSTREAM
           | elements EOSTREAM

// elements is a sequence of one or more elements
elements   : element
           | elements element

// an element is either short or long format
element    : shortformat
           | longformat

shortformat: '<' IDENTIFIER            '/>'
           | '<' IDENTIFIER attributes '/>'

longformat : head tail
           | head content tail

head       : '<' IDENTIFIER '>'
           | '<' IDENTIFIER attributes '>'

tail       : '</' IDENTIFIER '>'

// content is either verbatim or elements
content    : verbatim
           | elements

verbatim   : sequence
           | verbatim sequence

sequence   : TEXT
           | dereference

// attributes is a sequence of attribute
attributes : attribute
           | attributes attribute

attribute  : IDENTIFIER '=' value

value      : INTEGRAL
           | REAL
	   | dereference
	   | STRING compound

compound   : simple
           | compound simple

simple     : STRING
           | dereference

dereference: '${' IDENTIFIER '}'


TERMINALS:
----------
<
> 
/>
</
=
${
}
IDENTIFIER := [A-Za-z][A-Za-z0-9]*
STRING     := '\'' (.*) '\''
INTEGRAL   := [1-9][0-9]*
REAL       := [1-9][0-9]* (.[0-9]*)?
TEXT       := .*
EOSTREAM
